  <script>
    const pineScriptCode = `//@version=5
indicator("CJ App: 4 Variations Market Structure", overlay=true, max_boxes_count=100)

// --- Settings ---
grp_sessions = "Session Times (UTC/Exchange Time)"
asia_session = input.session("0000-0800", "Asia Session", group=grp_sessions)
london_session = input.session("0800-1300", "London Session", group=grp_sessions)
ny_session = input.session("1300-2000", "New York Session", group=grp_sessions)

// Colors
color_asia = color.new(color.blue, 90)
color_london = color.new(color.orange, 90)
color_ny = color.new(color.green, 90)

// --- Session Logic ---
in_session(sess) =>
    not na(time(timeframe.period, sess))

is_asia   = in_session(asia_session)
is_london = in_session(london_session)
is_ny     = in_session(ny_session)

// Track Highs and Lows
var float asia_hi = na
var float asia_lo = na
var float london_hi = na
var float london_lo = na
var float ny_hi = na
var float ny_lo = na

// Asia box drawing variables
var box asia_box = na

if is_asia and not is_asia[1]
    asia_hi := high
    asia_lo := low
    asia_box := box.new(left=bar_index, top=asia_hi, right=bar_index, bottom=asia_lo, border_color=color.blue, bgcolor=color_asia)
else if is_asia
    asia_hi := math.max(asia_hi, high)
    asia_lo := math.min(asia_lo, low)
    box.set_top(asia_box, asia_hi)
    box.set_bottom(asia_box, asia_lo)
    box.set_right(asia_box, bar_index)

if is_london and not is_london[1]
    london_hi := high
    london_lo := low
else if is_london
    london_hi := math.max(london_hi, high)
    london_lo := math.min(london_lo, low)

if is_ny and not is_ny[1]
    ny_hi := high
    ny_lo := low
else if is_ny
    ny_hi := math.max(ny_hi, high)
    ny_lo := math.min(ny_lo, low)

// Show Lines for Asia High/Low during London/NY
plot(not is_asia and asia_hi ? asia_hi : na, title="Asia High", color=color.blue, style=plot.style_linebr, linewidth=2)
plot(not is_asia and asia_lo ? asia_lo : na, title="Asia Low", color=color.blue, style=plot.style_linebr, linewidth=2)

// --- Variation Detection Logic ---
var int current_variation = 0

london_breaks_asia_high = london_hi > asia_hi
london_breaks_asia_low  = london_lo < asia_lo

ny_breaks_london_high = ny_hi > london_hi
ny_breaks_london_low  = ny_lo < london_lo

if is_ny and not is_ny[1]
    if london_breaks_asia_high
        current_variation := 1
    else if london_breaks_asia_low
        current_variation := 3
        
if is_ny
    if current_variation == 1 or current_variation == 2
        if ny_breaks_london_high
            current_variation := 1
        else
            current_variation := 2
            
    if current_variation == 3 or current_variation == 4
        if ny_breaks_london_low
            current_variation := 3
        else
            current_variation := 4

// --- Candlestick Patterns (Basic Bull/Bear Engulfing for entry) ---
bull_engulfing = open[1] > close[1] and close > open and close > open[1] and open < close[1]
bear_engulfing = open[1] < close[1] and close < open and close < open[1] and open > close[1]

// Signals
sell_signal = is_ny and (current_variation == 1 or current_variation == 2) and bear_engulfing
buy_signal  = is_ny and (current_variation == 3 or current_variation == 4) and bull_engulfing

plotshape(sell_signal, title="Sell Signal", style=shape.triangledown, location=location.abovebar, color=color.red, size=size.small, text="SELL (Var 1/2)")
plotshape(buy_signal, title="Buy Signal", style=shape.triangleup, location=location.belowbar, color=color.green, size=size.small, text="BUY (Var 3/4)")`;

    function copyStrategy() {
      const btn = document.querySelector('.copy-btn');
      const originalText = btn.innerText;

      navigator.clipboard.writeText(pineScriptCode).then(() => {
        btn.innerText = "✅ Copied Successfully!";

      });
    }

    function switchView(viewId) {
      const buttons = document.querySelectorAll('.nav-link:not(.copy-btn)');
      buttons.forEach(btn => {
        if (btn.getAttribute('onclick') && btn.getAttribute('onclick').includes(viewId)) {
          btn.classList.add('active');
        } else {
          btn.classList.remove('active');
        }
      });

      const sections = document.querySelectorAll('.view-section');
      sections.forEach(sec => {
        if (sec.id === 'view-' + viewId) {
          sec.classList.add('active');
        } else {
          sec.classList.remove('active');
        }
      });
    }

    function checkExpiration() {
      const authTime = localStorage.getItem('cj_app_auth');
      if (!authTime) return false;

      const sessionStart = parseInt(authTime, 10);
      const currentTime = new Date().getTime();
      const hours24 = 24 * 60 * 60 * 1000;

      // If 24 hours naturally passed
      if (currentTime - sessionStart > hours24) {
        return true;
      }
      return false;
    }

      if (u === "Ádmyn" && p === "Nymda0226!") {
